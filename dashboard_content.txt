<script>
let refreshTimer;

async function api(path) {
  const r = await fetch(path);
  const j = await r.json();
  if (!j.success) throw new Error(JSON.stringify(j));
  return j;
}

async function refreshAll() {
  try {
    const [gpus, tasks, status, policy] = await Promise.all([
      api('/api/gpus'),
      api('/api/tasks'),
      api('/api/scheduler/status'),
      api('/api/scheduler/policy'),
    ]);
    renderStats(status);
    renderGPUs(gpus.gpus);
    renderTasks(tasks.tasks);
    renderPolicy(policy);
  } catch(e) {
    console.error(e);
  }
}

function renderStats(s) {
  document.getElementById('stat-gpus').textContent = s.total_gpus;
  document.getElementById('stat-running').textContent = s.running_tasks;
  document.getElementById('stat-pending').textContent = s.pending_tasks;
  document.getElementById('stat-policy').textContent = s.policy;
}

function renderGPUs(gpus) {
  const grid = document.getElementById('gpu-grid');
  grid.innerHTML = '';
  gpus.forEach(g => {
    const isNvidia = g.Info.Type === 'nvidia';
    const vramPct = ((g.Usage.UsedVRAMMB || 0) / g.Info.VRAMMB * 100).toFixed(1);
    const card = document.createElement('div');
    card.className = 'gpu-card ' + (isNvidia ? 'gpu-nvidia' : 'gpu-amd');
    card.innerHTML = `
      <div class="gpu-name">${g.Info.Name}</div>
      <div class="gpu-type ${isNvidia ? 'nvidia' : 'amd'}">${g.Info.Type.toUpperCase()} | ${g.Info.VRAMMB} MB VRAM</div>
      <div class="vram-bar"><div class="vram-bar-fill ${isNvidia ? 'nvidia' : 'amd'}" style="width:${vramPct}%"></div></div>
      <div style="font-size:12px;color:#888;margin-bottom:12px;">VRAM: ${g.Usage.UsedVRAMMB || 0} / ${g.Info.VRAMMB} MB (${vramPct}%)</div>
      <div class="gpu-stats">
        <span class="gpu-stat-label">Running Tasks</span><span class="gpu-stat-value">${g.Usage.RunningTasks || 0}</span>
        <span class="gpu-stat-label">Utilization</span><span class="gpu-stat-value">${g.Usage.UtilizationPct || 0}%</span>
        <span class="gpu-stat-label">Temperature</span><span class="gpu-stat-value">${g.Usage.TemperatureC || '-'}°C</span>
        <span class="gpu-stat-label">Power Draw</span><span class="gpu-stat-value">${g.Usage.PowerDrawW || '-'}W</span>
      </div>
    `;
    grid.appendChild(card);
  });
}

function renderTasks(tasks) {
  const tbody = document.getElementById('task-tbody');
  if (!tasks || tasks.length === 0) {
    tbody.innerHTML = '<tr><td colspan="7" style="padding:20px;color:#666;">No tasks yet</td></tr>';
    return;
  }
  tbody.innerHTML = '';
  tasks.forEach(t => {
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td style="color:#00d4ff;">${t.ID}</td>
      <td>${t.Name}</td>
      <td><span class="badge badge-${t.Type}">${t.Type}</span></td>
      <td>${t.Priority}</td>
      <td>${t.GPUReq >= 0 ? 'GPU ' + t.GPUReq : '-'}</td>
      <td><span class="badge badge-${t.Status}">${t.Status}</span></td>
      <td>${t.VRAMMB || '-'}</td>
    `;
    tbody.appendChild(tr);
  });
}

function renderPolicy(p) {
  document.getElementById('stat-policy').textContent = p.policy;
  document.getElementById('policy-desc').textContent = p.description || '';
  ['binpack','spread','gpu_type'].forEach(k => {
    document.getElementById('btn-' + k).classList.toggle('active', k === p.policy);
  });
}

async function setPolicy(policy) {
  try {
    await fetch('/api/scheduler/policy', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({policy})
    });
    refreshAll();
  } catch(e) { console.error(e); }
}

// Auto-refresh every 5s
refreshAll();
refreshTimer = setInterval(refreshAll, 5000);
